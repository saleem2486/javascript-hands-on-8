Q2:Using Map get the count of alphabets

        let str="abcadeecfb";
        const m=new Map();

        for(let i=0;i < str.length;i++)
        {
          if(m.has(str[i]))
          {
            m.set(str[i],Number(m.get(str[i]))+1)
          }
          else
          {
            m.set(str[i],1);
          }
        }
        for(const[key,value] of m)
        {
          console.log(key+"="+value)
        }
        m.forEach((value,key)=>
        {
          console.log(key+"="+value);
        });
